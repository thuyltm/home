# pkg-config --variable=pc_path pkg-config
find_package(PkgConfig)
if (PKGCONFIG_FOUND)
    pkg_check_modules(GSTREAMER REQUIRED gstreamer-1.0 gstreamer-video-1.0)
    if (GSTREAMER_FOUND)
        add_executable(tutorial-1 tutorial-1.c)
        target_include_directories(tutorial-1 PRIVATE ${GSTREAMER_INCLUDE_DIRS})
        target_link_libraries(tutorial-1 PRIVATE ${GSTREAMER_LIBRARIES})
    else()
        message(FATAL_ERROR "SDL2 not found via pkg-config.")
    endif()
else()
    message(FATAL_ERROR "PkgConfig not found. Please install pkg-config.")
endif()
